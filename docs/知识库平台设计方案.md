### 2.2 数据流设计

```
写操作流程:
EduSmart Web App → HTTP POST → Backend API → fs.writeFile() → Obsidian Vault (.md)
                         ↓
                    ChromaDB Upsert → 向量索引更新

读操作流程:
EduSmart Web App → HTTP GET → Backend API → fs.readFile() → Obsidian Vault
                                      ↓ (RAG场景)
                              ChromaDB Query → 向量检索 → 相关片段
                                      ↓
                              Local LLM → 生成回答
```

### 3.1 同步架构

```

│  └────────┬────────┘ │
│           ▼           │
│  ┌─────────────────┐ │
│  │ Sync Executor    │ │ ← 执行同步任务
│  └────────┬────────┘ │
└───────────┼───────────┘
```

### 3.2 同步策略

#### 3.2.1 全量同步（初始化）

首次部署或索引重建时执行：

```javascript
// sync-manager.js
async function fullSync() {
  const vaultPath = 'C:\\Users\\lenovo\\Documents\\Obsidian Vault';
  
  // 1. 扫描所有 .md 文件
  const files = await scanMarkdownFiles(vaultPath);
  
  // 2. 解析 Frontmatter + 内容
  const documents = await Promise.all(
    files.map(f => parseMarkdownFile(f))
  );
  
  // 3. 批量写入 MySQL（元数据）
  await db.batchInsert('knowledge_notes', documents);
  
  // 4. 分块并向量化写入 ChromaDB
  const chunks = chunkDocuments(documents, { maxChunkSize: 512 });
  await chromaDB.upsert(chunks);
}
```

#### 3.2.2 增量同步（运行时）

```javascript
const chokidar = require('chokidar');

const watcher = chokidar.watch(vaultPath, {
  ignored: /(^|[\/\\])\../, // 忽略隐藏文件
  persistent: true
});

watcher.on('change', debounce(async (filePath) => {
  const doc = await parseMarkdownFile(filePath);
  await db.upsert('knowledge_notes', doc);
  await chromaDB.upsertDocument(doc);
}, 2000)); // 2秒防抖

watcher.on('unlink', async (filePath) => {
  const noteId = pathToId(filePath);
  await db.delete('knowledge_notes', noteId);
  await chromaDB.delete(noteId);
});
```

#### 3.2.3 同步触发条件

| 触发方式 | 场景 | 延迟 |
|----------|------|------|
| 文件变更事件 | 用户在 Obsidian 中编辑笔记 | 2秒防抖 |
| API 写入 | 前端通过 API 写入/更新笔记 | 即时 |
| 定时任务 | 定期全量校验一致性 | 每24小时 |
| 手动触发 | 管理员主动触发重建 | 即时 |

```sql
  id INT PRIMARY KEY AUTO_INCREMENT,
  note_id VARCHAR(255) UNIQUE NOT NULL,      -- 唯一标识 (文件路径hash)

  INDEX idx_category (category),
  INDEX idx_sub_category (sub_category),
  INDEX idx_tags (tags),
```

### 5.2 RAG 后端实现

```javascript
// api/rag.js

async function ragQuery(question, options = {}) {
  const {
    topK = 5,
    category = null,      // 可选分类过滤
    minSimilarity = 0.6,  // 最低相似度阈值
  } = options;

  // 1. 向量检索
  const chromaClient = new ChromaClient();
  const collection = await chromaClient.getCollection('edusmart_knowledge_base');
  
  const results = await collection.query({
    queryTexts: [question],
    nResults: topK * 2,  // 取2倍用于重排
    where: category ? { category } : undefined,
  });

  // 2. 结果重排与过滤
  const reranked = results.documents[0]
    .map((doc, i) => ({
      doc,
      metadata: results.metadatas[0][i],
      distance: results.distances[0][i],
      similarity: 1 - results.distances[0][i],
    }))
    .filter(r => r.similarity >= minSimilarity)
    .sort((a, b) => b.similarity - a.similarity)
    .slice(0, topK);

  // 3. 组装 Prompt
  const context = reranked.map(r => 
    `[来源: ${r.metadata.title}]\n${r.doc}`
  ).join('\n\n---\n\n');
  
  const prompt = buildRagPrompt(question, context);

  // 4. 调用本地 LLM
  const response = await llmGateway.chat({
    model: process.env.LOCAL_LLM_MODEL,
    messages: [
      { role: 'system', content: '你是 EduSmart 智能学习助手。基于知识库内容回答用户问题。' },
      { role: 'user', content: prompt }
    ],
    temperature: 0.4,
    max_tokens: 2048,
  });

  // 5. 返回结果
  return {
    answer: response.content,
    sources: reranked.map(r => ({
      title: r.metadata.title,
      path: r.metadata.file_path,
      sub_category: r.metadata.sub_category,
      excerpt: r.doc.substring(0, 200) + '...',
      similarity: r.similarity,
    })),
    related: await getRelatedQuestions(question),
  };
}
```

```javascript
// 基于所有笔记构建知识图谱
async function buildFullKnowledgeGraph() {
  const allNotes = await db.query('SELECT * FROM knowledge_notes');
  
  const graph = {
    nodes: [],
    edges: [],
  };
  
  for (const note of allNotes) {
    // 添加节点
    graph.nodes.push({
      id: note.note_id,
      label: note.title,
      category: note.category,
      sub_category: note.sub_category,
      url: note.source_url,
    });
    
    // 提取内部链接（[[双向链接]]）
    const links = extractWikiLinks(note.content);
    for (const link of links) {
      graph.edges.push({
        source: note.note_id,
        target: link,
        type: 'reference',
      });
    }
    
    // 提取前置依赖关系
    const prerequisites = await extractPrerequisites(note.content);
    for (const prereq of prerequisites) {
      graph.edges.push({
        source: prereq,
        target: note.note_id,
        type: 'prerequisite',
      });
    }
  }
  
  return graph;
}
```

```javascript
async function askRagQuestion() {
  const input = document.getElementById('rag-input');
  const question = input.value.trim();
  if (!question) return;
  
  // 显示用户问题
  appendMessage('user', question);
  input.value = '';
  
  // 显示加载状态
  const loadingId = appendLoading();
  
  // 调用 RAG API
```
